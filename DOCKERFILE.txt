FROM mcr.microsoft.com/dotnet/sdk:3.1

  apt-get install -y --no-install-recommends python3 python3-distutils && \

  ln -sf /usr/bin/python3 /usr/bin/python

ENV PATH="$PATH:/root/.dotnet/tools"
